/**
 * A class that maintains information on a book.
 * This might form part of a larger application such
 * as a library system, for instance.
 *
 * @author (Insert your name here.)
 * @version (Insert today's date here.)
 */
class Book
{
    // The fields.
    private String author;
    private String title;
    private int pages;
    private String refNumber;
    private int borrowed;

    /**
     * Set the author and title fields when this object
     * is constructed.
     */
    public Book(String bookAuthor, String bookTitle, int bookPages)
    {
        author = bookAuthor;
        title = bookTitle;
        pages = bookPages;
        refNumber = "";
        borrowed = 0;
    }

    /**
     * get author
     */
    public String getAuthor()
    {
    
    return(author);
    
    }
    
    /**
     * get title
     */
    public String getTitle()
    {
    
    return(title);    
    
    }
    
    /**
     * get number of pages
     */
    
    public int getPages()
    {
     
    return(pages);    
    }    
    
    
    /**
     * print author name
     */
    public void printAuthor()
    {
     
    System.out.println("The Author's name is: " + author);
        
    }
    
    
    /**
     * print book title
     */
    
    public void printTitle()
    {
    
    System.out.println("The title of the book is: " + title);
    
    }
    
    
    /**
     * print details of a given book
     */
    
    public void printDetails()
    {
    
    System.out.println("Title: " + title);
    System.out.println("Author: " + author);
    System.out.println("No. of Pages: " + pages);
    if(refNumber.length() >=1) {
        System.out.println("Reference Number: " + refNumber);
    } else {
        System.out.println("Reference Number: ZZZ");
    }
    System.out.println("The book has been borrowed, " + borrowed + " times");
    }
    
    /**
     * get reference number
     */
    
    public String getRefNumber()
    {
    return(refNumber); 
    }
    
    /**
     * set reference number
     */
    
    public void setRefNummber(String ref)
    {
    if(ref.length() >=3) {
        refNumber = ref;
    } else {
        refNumber="";
        System.out.println("The reference number is too short, and is therefore set to empty!");
    }
    }
    
    
    /**
     * update borrowed counter
     */
    
    public void setBorrowed()
    {
    borrowed += 1;
    }
    
}
